This file is the single source of truth for this Emacs setup. Read it top to bottom for boot order: early variables and paths run first, then UI and editing conveniences, then navigation and completion, version control, programming support, Org, utilities, and finally local =lisp/= modules. Code blocks are Emacs Lisp; tangling (if you use it) should mirror that order so hooks and =require= calls see the dependencies they expect.

* CORE
Everything in this chapter loads or defines shared state that later sections rely on: font names, module paths, faces themes can override, and baseline =use-package= / built-in tweaks. Treat it as the “prelude” before optional UI and language-specific packages.

** VARIABLES
Names and paths used throughout the config, plus early =require= of tidy and icon helpers so redirected data directories exist before packages write cache or state. Also defines TODO-style faces and neutral “liminal” semantic faces so themes and font-lock stay consistent instead of hard-coding colors in each block.

** TREESITTER
Grammar sources are declared eagerly so ~treesit-install-language-grammar~ always has recipes available, regardless of which modes have been activated.

#+begin_src emacs-lisp
  (setq treesit-language-source-alist
	'((bash       . ("https://github.com/tree-sitter/tree-sitter-bash"))
	  (dockerfile . ("https://github.com/camdencheek/tree-sitter-dockerfile"))
	  (javascript . ("https://github.com/tree-sitter/tree-sitter-javascript"))
	  (jsdoc      . ("https://github.com/tree-sitter/tree-sitter-jsdoc"))
	  (ruby       . ("https://github.com/tree-sitter/tree-sitter-ruby"))
	  (tsx        . ("https://github.com/tree-sitter/tree-sitter-typescript" "master" "tsx/src"))
	  (typescript . ("https://github.com/tree-sitter/tree-sitter-typescript" "master" "typescript/src"))
	  (yaml       . ("https://github.com/tree-sitter-grammars/tree-sitter-yaml"))))

  ;; Remap legacy modes to their tree-sitter equivalents.
  ;; This is the correct Emacs 29+ mechanism — takes priority over auto-mode-alist.
  (dolist (mapping '((sh-mode          . bash-ts-mode)
		     (ruby-mode         . ruby-ts-mode)
		     (js-mode           . js-ts-mode)
		     (typescript-mode   . typescript-ts-mode)
		     (yaml-mode         . yaml-ts-mode)
		     (dockerfile-mode   . dockerfile-ts-mode)))
    (add-to-list 'major-mode-remap-alist mapping))

  (defun mutecipher/treesit-install-all-grammars ()
    "Install missing grammars from `treesit-language-source-alist' to `mutecipher/data-dir'.
Skips any language whose grammar is already available."
    (interactive)
    (let ((out-dir (expand-file-name "tree-sitter/" mutecipher/data-dir)))
      (make-directory out-dir t)
      (dolist (lang (mapcar #'car treesit-language-source-alist))
	(if (treesit-language-available-p lang)
	    (message "Tree-sitter: %s already installed, skipping" lang)
	  ;; Emacs 30+ accepts an explicit out-dir argument; Emacs 29 does not.
	  (if (>= emacs-major-version 30)
	      (treesit-install-language-grammar lang out-dir)
	    (let ((user-emacs-directory mutecipher/data-dir))
	      (treesit-install-language-grammar lang)))))
      (message "Tree-sitter grammars installed to %s" out-dir)))

** BLOG
Creates a new blog post file in =~/Documents/Blog/= with pre-filled Org metadata; invoke with =C-c b n=.

#+begin_src emacs-lisp
  (defcustom mutecipher-blog-directory (expand-file-name "~/Documents/Blog/")
    "Directory where blog post Org files are stored."
    :type 'directory
    :group 'mutecipher)

  (defun mutecipher--blog-slugify (title)
    "Convert TITLE to a URL-safe slug."
    (thread-last title
      (downcase)
      (replace-regexp-in-string "[^a-z0-9]+" "-")
      (replace-regexp-in-string "^-\\|-$" "")))

  (defun mutecipher/new-blog-post (title)
    "Create a new blog post Org file in `mutecipher-blog-directory'.
Prompts for TITLE (required) then all supported metadata fields."
    (interactive "sBlog post title: ")
    (let* ((slug-derived  (mutecipher--blog-slugify title))
           (date          (format-time-string "%Y-%m-%d"))
           (filename      (format "%s-%s.org" date slug-derived))
           (filepath      (expand-file-name filename mutecipher-blog-directory))
           (desc          (read-string "Description (optional): "))
           (tags-raw      (read-string "Tags, space-separated (optional): "))
           (tags          (unless (string-empty-p tags-raw)
                            (concat ":"
                                    (mapconcat #'identity (split-string tags-raw) ":")
                                    ":")))
           (image         (read-string "Hero image path or URL (optional): "))
           (image-alt     (unless (string-empty-p image)
                            (read-string "Hero image alt text (optional): ")))
           (slug-override (read-string (format "URL slug [%s]: " slug-derived)))
           (slug          (if (string-empty-p slug-override) slug-derived slug-override))
           (draft         (y-or-n-p "Save as draft? ")))
      (unless (file-directory-p mutecipher-blog-directory)
        (make-directory mutecipher-blog-directory t))
      (find-file filepath)
      (insert "#+TITLE:       " title "\n")
      (insert "#+DATE:        " date "\n")
      (insert "#+AUTHOR:      " (or user-full-name "") "\n")
      (insert "#+DESCRIPTION: " desc "\n")
      (insert "#+FILETAGS:    " (or tags "") "\n")
      (insert "#+IMAGE:       " image "\n")
      (insert "#+IMAGE_ALT:   " (or image-alt "") "\n")
      (insert "#+SLUG:        " slug "\n")
      (insert "#+DRAFT:       " (if draft "true" "false") "\n")
      (insert "\n")
      (save-buffer)))

  (global-set-key (kbd "C-c b n") #'mutecipher/new-blog-post)
#+end_src
